void setup()
{
    size(621, 580);
    PImage startImage = loadImage( "0.png" );
    image( startImage, 0, 0 );
    
    for( int i = 0; i < NUM_OF_CHORDS; i++ )

    initializeChords();
    initializeAnimation();

void draw()
{

        image( secondImage, 0, 0 );
    }
}
